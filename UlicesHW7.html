for (df in bostonDF) 
{
  if (df > 1) 
  {
    numKnots <- df - 1
    bostonKnots <- quantile(boston$dis, probs = seq(0, 1, length.out = numKnots + 2)[2:(numKnots + 1)])
    bostonSpline <- lm(nox ~ bs(dis, knots = bostonKnots, degree = 3), data = boston)
  } 
  else 
  {
    bostonSpline <- lm(nox ~ bs(dis, degree = 3, df = df), data = boston)
